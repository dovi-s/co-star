Yes. Do it as a single “Video Rehearsal” mode that turns the entire page into a live camera feed with the reader UI overlaid. You read while the app records you, hands-free.

Below is a working browser prototype (one file). It:

Fills the page with your camera, FaceTime-style.

Overlays the three-line reader and transport.

Records a composite of camera + overlay to a single WebM file.

Records your mic (so coaches hear you).
Note: standard browsers cannot reliably capture Web Speech (TTS) audio into the same recording. In native apps we’ll mix partner audio; on the web you get your voice + video + overlay. Everything else behaves the same.

Copy-paste as video-rehearsal.html and open in Chrome (desktop or Android). iOS Safari works for live mode; recording requires user gesture permissions.

async function startCamera(){
  try{
    camStream = await navigator.mediaDevices.getUserMedia({
      video: {facingMode: useFront ? "user" : "environment"},
      audio: true
    });
    videoEl.srcObject = camStream;
    await videoEl.play();
    document.getElementById('camChip').textContent = "Camera • on";
    document.getElementById('micChip').textContent = "Mic • on";
    drawLoop();
  }catch(e){
    alert("Camera/Mic permission denied.");
  }
}
function stopCamera(){
  camStream?.getTracks().forEach(t=>t.stop());
  camStream = null;
  document.getElementById('camChip').textContent = "Camera • off";
  document.getElementById('micChip').textContent = "Mic • off";
}
function resizeCanvas(){
  const dpr = Math.min(window.devicePixelRatio||1, 2);
  const {clientWidth:w, clientHeight:h} = document.getElementById('stageWrap');
  canvas.width = Math.floor(w*dpr); canvas.height = Math.floor(h*dpr);
  canvas.style.width = w+"px"; canvas.style.height = h+"px";
}
window.addEventListener('resize', resizeCanvas); resizeCanvas();

function drawLoop(){
  if(!videoEl.videoWidth){ requestAnimationFrame(drawLoop); return; }
  // fill canvas with camera, cover mode
  const cw = canvas.width, ch = canvas.height;
  const vw = videoEl.videoWidth, vh = videoEl.videoHeight;
  const scale = Math.max(cw/vw, ch/vh);
  const dw = vw*scale, dh = vh*scale;
  const dx = (cw - dw)/2, dy = (ch - dh)/2;
  ctx.drawImage(videoEl, dx, dy, dw, dh);
  // subtle vignette
  const grd = ctx.createRadialGradient(cw/2,ch/2,Math.min(cw,ch)/3, cw/2,ch/2, Math.max(cw,ch)/1.05);
  grd.addColorStop(0, "rgba(0,0,0,0)"); grd.addColorStop(1, "rgba(0,0,0,.35)");
  ctx.fillStyle = grd; ctx.fillRect(0,0,cw,ch);
  // draw overlay text (active line baked into recording)
  drawOverlayIntoCanvas();
  requestAnimationFrame(drawLoop);
}

function drawOverlayIntoCanvas(){
  const pad = 24, w = canvas.width, h = canvas.height;
  const cardW = Math.min(w - pad*2, w*0.92);
  const cardH = 150 * Math.min(window.devicePixelRatio||1,2);
  const y = h - cardH - 220;
  // Card
  roundRect(ctx, (w-cardW)/2, y, cardW, cardH, 16, "#0B1222cc");
  ctx.strokeStyle = "rgba(255,255,255,.15)"; ctx.lineWidth = 1; ctx.stroke();
  // Active line text
  ctx.fillStyle = "#fff";
  ctx.font = "700 " + (22*(window.devicePixelRatio||1)) + "px Inter, system-ui, sans-serif";
  wrapText(ctx, script[idx].who + ": " + script[idx].text, (w-cardW)/2+18, y+44, cardW-36, 30);
}

function roundRect(ctx,x,y,w,h,r,fill){
  ctx.beginPath();
  ctx.moveTo(x+r, y);
  ctx.arcTo(x+w, y, x+w, y+h, r);
  ctx.arcTo(x+w, y+h, x, y+h, r);
  ctx.arcTo(x, y+h, x, y, r);
  ctx.arcTo(x, y, x+w, y, r);
  ctx.closePath();
  ctx.fillStyle = fill; ctx.fill();
}
function wrapText(ctx, text, x, y, maxWidth, lineHeight){
  const words = text.split(' '); let line = '';
  for (let n=0; n<words.length; n++){
    const test = line + words[n] + ' ';
    const w = ctx.measureText(test).width;
    if (w > maxWidth && n>0){ ctx.fillText(line, x, y); line = words[n] + ' '; y += lineHeight; }
    else line = test;
  }
  ctx.fillText(line, x, y);
}

function render(){

}

let recorder, recordedBlobs=[], mixedStream=null, micSource=null, dest=null;
async function startRecording(){
  if(!camStream) await startCamera();
  // 1) Canvas stream (video with overlay baked in)
  const canvasStream = canvas.captureStream(30);
  // 2) Mic stream (from getUserMedia)
  const mic = camStream.getAudioTracks().length ? new MediaStream([camStream.getAudioTracks()[0]]) : null;

  // 3) Combine video + mic into one MediaStream
  mixedStream = new MediaStream();
  canvasStream.getVideoTracks().forEach(t=>mixedStream.addTrack(t));
  if(mic) mic.getAudioTracks().forEach(t=>mixedStream.addTrack(t));

  recordedBlobs = [];
  recorder = new MediaRecorder(mixedStream, {mimeType:'video/webm;codecs=vp9,opus'});
  recorder.ondataavailable = e=>{ if(e.data && e.data.size) recordedBlobs.push(e.data); };
  recorder.onstop = ()=>{
    const blob = new Blob(recordedBlobs, {type:'video/webm'});
    const url = URL.createObjectURL(blob);
    dlBtn.disabled=false; dlBtn.href = url; dlBtn.download = `castmate-rehearsal-${Date.now()}.webm`;
    document.getElementById('recChip').textContent = "Stopped";
  };
  recorder.start(1000);
  document.getElementById('recChip').textContent = "Recording…";
}
function stopRecording(){
  recorder?.stop();
  mixedStream?.getTracks().forEach(t=>t.stop());
}

const stopBtn = document.getElementById('stopBtn');

Notes and next steps

Browser limitation: mixing system TTS into the recording is unreliable. In native apps we’ll mix partner audio with your mic in the same file.

For production, the overlay text should be rendered via the same prosody planner used for speaking (so what you see equals what you hear).

Add a “Safe Mode” rule: if motion/Bluetooth car is detected, hide video controls and go audio-only; keep the same transport.

Add a small “Preview frame” guide (framing grid, mic meter) before recording begins; this file already includes a progress meter and status chips.

This achieves the FaceTime-style “full video with reader overlay while recording” you described, with minimal controls and the same Start/Pause/Next loop.

function wrapText(ctx, text, x, y, maxW, lh){
  const words=text.split(' '); let line='';
  for(let n=0;n<words.length;n++){
    const t=line+words[n]+' ';
    if(ctx.measureText(t).width>maxW && n){ ctx.fillText(line,x,y); line=words[n]+' '; y+=lh; }
    else line=t;
  } ctx.fillText(line,x,y);
}
